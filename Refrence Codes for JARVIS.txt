4: For sending message on WhatsApp:
import pyautogui
import time
pyautogui.press('win')
pyautogui.typewrite('whatsapp')
pyautogui.press('enter')
time.sleep(1)
pyautogui.hotkey('ctrl', 'f')
pyautogui.typewrite('{Contact Name}')
time.sleep(0.5)
pyautogui.press('enter')
pyautogui.typewrite("{message}")
time.sleep(0.5)
pyautogui.press('enter')

5: For WhatsApp call:
import pyautogui
import time
pyautogui.press('win')
pyautogui.typewrite('whatsapp')
pyautogui.press('enter')
time.sleep(1)
pyautogui.hotkey('ctrl', 'f')
pyautogui.typewrite('{Contact Name}')
time.sleep(1)
call_type = 'voice'
pyautogui.press('enter')
if call_type.lower() == "voice":
    pyautogui.click(1747, 82)
else:
    pyautogui.click(1682, 76)
